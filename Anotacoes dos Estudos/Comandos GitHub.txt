COMANDOS
mkdir - CRIA UM REPOSITORIO
git clone https://github.com/valdemirsm/repositoryteste2.git - COPIA O REPOSITORIO PARA A MAQUINA LOCAL
git tag -l - LISTA AS VERSÕES EXISTENTES DENTRO DA PASTA TAGS
git checkout v0.1 - ABAIXA PARA A MAQUILA LOCA A VERSÃO V0.1 QUE ESTA DENTRO DA PASTA TAGS
git diff v0.1 v0.2 - LISTA AS DIFERENÇA NOS ARQUIVOS ENTRE AS DUAS VERSÕES
git blame index.html - MOSTRA LINHA A LINHA QUEM FEZ ALTERAÇÃO.
git status - LISTA OS AQUIVO QUE TEVE ALTERAÇÃO
git add index.html - ADICIONA O ARQUIVO ALTERADO PARA SER COMITADO.
git commit -m "Arquivo adicionado" - COMITA O ARQUIVO ALTERADO.
git push - ENVIA OS ARQUIVOS COMITATOS PARA O REPOSITORIO REMOTO
git log - LISTA O HISTORICO DE COMITS
git pull - LISTA OS ARQUIVOS QUE TIVERÃO ALTERAÇÃO E QUE ESTÃO NO REPOSITORIO REMOTO, ACONTECE QUANTO TEM DOIS USER TRABALHANDO
